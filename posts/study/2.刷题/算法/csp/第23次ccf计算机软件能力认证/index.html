const int N = 500005, M = 10005;
int a[N], cnt[M];
int n;

int main(){
    cin>>n;
    for(int i = 1; i <= n; i++){
        cin>>a[i];
    }
    n = unique(a + 1, a + n + 1) - a - 1;
    a[0] = a[n + 1] = 0;
    
    for(int i = 1; i <= n; i++){
        if(a[i - 1] < a[i] && a[i + 1] < a[i]) cnt[a[i]]++;
        else if(a[i - 1] > a[i] && a[i + 1] > a[i]) cnt[a[i]]--;
    }
    
    int res = 0, sum = 0;
    for(int i = M - 1; i; i--){
        sum += cnt[i];
        res = max(res, sum);
    }
    cout<<res;
    
    return 0;
}
