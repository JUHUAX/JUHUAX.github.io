邻接表

Posted on Jun 22, 2023

模板代码

int idx；
int h[N], e[N], ne[N], w[N;]
// 注意这里要初始化 head 为 -1
memset(h, 0xff, sizeof(h));
// 加入有向边 (x, y)，权值为 z
void add(int x, int y, int z) {
    // 真实数据
    e[idx] = y, w[idx] = z;
    // 在表头 x 处插入
    ne[idx] = h[x], h[x] = idx++;
}
// 访问从 x 出发的所有边，注意这里终止条件的不同
for (int i = h[x]; ~i; i = ne[i]) {//~i表示i不等于-1
    int y = e[i], z = w[i];
}
