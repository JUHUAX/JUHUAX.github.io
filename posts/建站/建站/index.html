[markup]
    defaultMarkdownHandler = "goldmark"
	
  [markup.goldmark]
    [markup.goldmark.renderer]
      unsafe = true

将上述参数添加到配置文件中，hugo将支持使用<br/>换行 然后markdown的换行是两个空格+回车。这样转成html才会是换行。
